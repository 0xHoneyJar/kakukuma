12/// Return the visual background color for an empty/transparent cell position.
13fn grid_bg(x: usize, y: usize, show_grid: bool) -> Color {
14    if show_grid {
15        if (x + y).is_multiple_of(2) {
16            Color::Indexed(236)
17        } else {
18            Color::Indexed(234)
19        }
20    } else {
21        Color::Reset
22    }
23}

25/// Resolve a half-block cell into (char, fg, bg) for terminal rendering,
26/// treating Color256::BLACK halves as transparent (grid/background).
27fn resolve_half_block(cell: Cell, x: usize, y: usize, show_grid: bool) -> (char, Color, Color) {
28    let canvas_bg = Color256::BLACK;
29
30    match cell.block {
31        BlockChar::UpperHalf => {
32            let top = cell.fg;
33            let bottom = cell.bg;
34            match (top == canvas_bg, bottom == canvas_bg) {
35                (true, true) => (' ', Color::Reset, grid_bg(x, y, show_grid)),
36                (false, true) => ('▀', top.to_ratatui(), grid_bg(x, y, show_grid)),
37                (true, false) => ('▄', bottom.to_ratatui(), grid_bg(x, y, show_grid)),
38                (false, false) => ('▀', top.to_ratatui(), bottom.to_ratatui()),
39            }
40        }
41        BlockChar::LeftHalf => {
42            let left = cell.fg;
43            let right = cell.bg;
44            match (left == canvas_bg, right == canvas_bg) {
45                (true, true) => (' ', Color::Reset, grid_bg(x, y, show_grid)),
46                (false, true) => ('▌', left.to_ratatui(), grid_bg(x, y, show_grid)),
47                (true, false) => ('▐', right.to_ratatui(), grid_bg(x, y, show_grid)),
48                (false, false) => ('▌', left.to_ratatui(), right.to_ratatui()),
49            }
50        }
51        // LowerHalf and RightHalf are non-canonical but handle defensively
52        BlockChar::LowerHalf => {
53            let bottom = cell.fg;
54            let top = cell.bg;
55            match (top == canvas_bg, bottom == canvas_bg) {
56                (true, true) => (' ', Color::Reset, grid_bg(x, y, show_grid)),
57                (false, true) => ('▀', top.to_ratatui(), grid_bg(x, y, show_grid)),
58                (true, false) => ('▄', bottom.to_ratatui(), grid_bg(x, y, show_grid)),
59                (false, false) => ('▄', bottom.to_ratatui(), top.to_ratatui()),
60            }
61        }
62        BlockChar::RightHalf => {
63            let right = cell.fg;
64            let left = cell.bg;
65            match (left == canvas_bg, right == canvas_bg) {
66                (true, true) => (' ', Color::Reset, grid_bg(x, y, show_grid)),
67                (false, true) => ('▌', left.to_ratatui(), grid_bg(x, y, show_grid)),
68                (true, false) => ('▐', right.to_ratatui(), grid_bg(x, y, show_grid)),
69                (false, false) => ('▐', right.to_ratatui(), left.to_ratatui()),
70            }
71        }
72        _ => unreachable!("resolve_half_block called for non-half-block"),
73    }
74}

76/// Render the canvas editor and return the screen area for mouse mapping.
77pub fn render(f: &mut Frame, app: &App, area: Rect) -> CanvasArea {
78    let canvas_w = (app.canvas.width * 2) as u16;
79    let canvas_h = app.canvas.height as u16;
80
81    // Center the canvas in the available area
82    let offset_x = (area.width.saturating_sub(canvas_w)) / 2;
83    let offset_y = (area.height.saturating_sub(canvas_h)) / 2;
84
85    let canvas_rect = Rect::new(
86        area.x + offset_x,
87        area.y + offset_y,
88        canvas_w.min(area.width),
89        canvas_h.min(area.height),
90    );
91
92    let widget = CanvasWidget { app };
93    f.render_widget(widget, canvas_rect);
94
95    CanvasArea {
96        left: canvas_rect.x,
97        top: canvas_rect.y,
98        width: canvas_rect.width,
99        height: canvas_rect.height,
100    }
101}

122struct CanvasWidget<'a> {
123    app: &'a App,
124}

126impl<'a> CanvasWidget<'a> {
127    fn is_in_tool_preview(&self, x: usize, y: usize) -> bool {
128        let cursor = match self.app.cursor {
129            Some(c) => c,
130            None => return false,
131        };
132        match &self.app.tool_state {
133            ToolState::LineStart { x: x0, y: y0 } => {
134                let points = tools::bresenham_line(*x0, *y0, cursor.0, cursor.1);
135                points.contains(&(x, y))
136            }
137            ToolState::RectStart { x: x0, y: y0 } => {
138                let min_x = (*x0).min(cursor.0);
139                let max_x = (*x0).max(cursor.0);
140                let min_y = (*y0).min(cursor.1);
141                let max_y = (*y0).max(cursor.1);
142                let is_border = x == min_x || x == max_x || y == min_y || y == max_y;
143                x >= min_x && x <= max_x && y >= min_y && y <= max_y && is_border
144            }
145            ToolState::Idle => false,
146        }
147    }
148}

150impl<'a> Widget for CanvasWidget<'a> {
151    fn render(self, area: Rect, buf: &mut Buffer) {
152        let cw = self.app.canvas.width;
153        let ch = self.app.canvas.height;
154        for y in 0..ch {
155            for x in 0..cw {
156                let screen_x = area.x + (x as u16) * 2;
157                let screen_y = area.y + y as u16;
158
159                if screen_x + 1 >= area.x + area.width || screen_y >= area.y + area.height {
160                    continue;
161                }
162
163                if let Some(cell) = self.app.canvas.get(x, y) {
164                    let is_cursor = self.app.cursor == Some((x, y));
165
166                    // Tool preview overlay (line/rect in progress)
167                    let render_cell = if self.is_in_tool_preview(x, y) && !is_cursor {
168                        tools::compose_cell(
169                            cell,
170                            self.app.active_block,
171                            self.app.color,
172                            Color256::BLACK,
173                        )
174                    } else {
175                        cell
176                    };
177
178                    // Symmetry axis check
179                    let on_h_axis = self.app.symmetry.has_horizontal()
180                        && (x == cw / 2 - 1 || x == cw / 2);
181                    let on_v_axis = self.app.symmetry.has_vertical()
182                        && (y == ch / 2 - 1 || y == ch / 2);
183                    let on_axis = (on_h_axis || on_v_axis) && !is_cursor;
184
185                    // Horizontal half-blocks (▌▐): render each terminal char
186                    // separately so the left/right split is clearly visible.
187                    let is_horiz_half = matches!(
188                        render_cell.block,
189                        BlockChar::LeftHalf | BlockChar::RightHalf
190                    );
191
192                    if is_horiz_half && !is_cursor {
193                        let (left, right) = match render_cell.block {
194                            BlockChar::LeftHalf => (render_cell.fg, render_cell.bg),
195                            _ => (render_cell.bg, render_cell.fg),
196                        };
197                        let canvas_bg = Color256::BLACK;
198                        let left_bg = if left == canvas_bg {
199                            grid_bg(x, y, true)
200                        } else {
201                            left.to_ratatui()
202                        };
203                        let right_bg = if right == canvas_bg {
204                            grid_bg(x, y, true)
205                        } else {
206                            right.to_ratatui()
207                        };
208                        buf.set_string(screen_x, screen_y, " ", Style::default().bg(left_bg));
209                        buf.set_string(screen_x + 1, screen_y, " ", Style::default().bg(right_bg));
210                    } else {
211                        let (ch, mut fg, mut bg) = match render_cell.block {
212                            BlockChar::Full => {
213                                let c = render_cell.fg.to_ratatui();
214                                (render_cell.block.to_char(), c, c)
215                            }
216                            BlockChar::Empty => {
217                                (' ', render_cell.fg.to_ratatui(), grid_bg(x, y, true))
218                            }
219                            _ => {
220                                // Vertical half-blocks, or horizontal under cursor
221                                resolve_half_block(render_cell, x, y, true)
222                            }
223                        };
224
225                        if on_axis && render_cell.block == BlockChar::Empty {
226                            bg = Color::Indexed(238);
227                        }
228
229                        if is_cursor {
230                            std::mem::swap(&mut fg, &mut bg);
231                        }
232
233                        let style = Style::default().fg(fg).bg(bg);
234                        let ch_str: String = std::iter::repeat_n(ch, 2).collect();
235                        buf.set_string(screen_x, screen_y, &ch_str, style);
236                    }
237                }
238            }
239        }
240    }
241}

243/// True terminal preview: 1 character per cell (actual output size).
244struct TruePreviewWidget<'a> {
245    app: &'a App,
246}
247
248impl<'a> Widget for TruePreviewWidget<'a> {
249    fn render(self, area: Rect, buf: &mut Buffer) {
250        for y in 0..self.app.canvas.height {
251            for x in 0..self.app.canvas.width {
252                let screen_x = area.x + x as u16;
253                let screen_y = area.y + y as u16;
254
255                if screen_x >= area.x + area.width || screen_y >= area.y + area.height {
256                    continue;
257                }
258
259                if let Some(cell) = self.app.canvas.get(x, y) {
260                    let (ch, fg, bg) = match cell.block {
261                        BlockChar::Full => {
262                            let c = cell.fg.to_ratatui();
263                            (cell.block.to_char(), c, c)
264                        }
265                        BlockChar::Empty => {
266                            (' ', cell.fg.to_ratatui(), Color::Reset)
267                        }
268                        BlockChar::UpperHalf | BlockChar::LowerHalf
269                        | BlockChar::LeftHalf | BlockChar::RightHalf => {
270                            resolve_half_block(cell, x, y, false)
271                        }
272                    };
273                    let style = Style::default().fg(fg).bg(bg);
274                    buf.set_string(screen_x, screen_y, ch.to_string(), style);
275                }
276            }
277        }
278    }
279}
