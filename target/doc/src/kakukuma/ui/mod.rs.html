15/// Render the full UI and return the canvas area for mouse mapping.
16pub fn render(f: &mut Frame, app: &App) -> CanvasArea {
17    let size = f.area();
18    let theme = app.theme();
19
20    // Check minimum size
21    if size.width < 100 || size.height < 36 {
22        let msg = Paragraph::new(format!(
23            "Terminal too small: {}x{}\nMinimum: 100x36\nPlease resize.",
24            size.width, size.height
25        ))
26        .style(Style::default().fg(Color::Red));
27        f.render_widget(msg, size);
28        return CanvasArea {
29            left: 0,
30            top: 0,
31            width: 0,
32            height: 0,
33        };
34    }
35
36    // Top-level: main bordered frame + status bar outside
37    let outer = Layout::default()
38        .direction(Direction::Vertical)
39        .constraints([
40            Constraint::Min(34),   // Main frame
41            Constraint::Length(1), // Status bar (outside border)
42        ])
43        .split(size);
44
45    let main_area = outer[0];
46    let status_area = outer[1];
47
48    // Render the main border frame
49    let main_block = Block::default()
50        .borders(Borders::ALL)
51        .border_type(BorderType::Rounded)
52        .border_style(Style::default().fg(theme.separator));
53    let inner = main_block.inner(main_area);
54    f.render_widget(main_block, main_area);
55
56    // Inside the frame: header + body
57    let vertical = Layout::default()
58        .direction(Direction::Vertical)
59        .constraints([
60            Constraint::Length(1), // Header
61            Constraint::Min(32),  // Body
62        ])
63        .split(inner);
64
65    let header_area = vertical[0];
66    let body_area = vertical[1];
67
68    // Header
69    render_header(f, app, header_area, theme);
70
71    // Body: left toolbar | canvas | right palette
72    let horizontal = Layout::default()
73        .direction(Direction::Horizontal)
74        .constraints([
75            Constraint::Length(14), // Toolbar (bordered panel)
76            Constraint::Min(64),   // Canvas (32 cells * 2 chars)
77            Constraint::Length(20), // Palette (bordered panel)
78        ])
79        .split(body_area);
80
81    let toolbar_area = horizontal[0];
82    let canvas_area = horizontal[1];
83    let palette_area = horizontal[2];
84
85    // Toolbar (3 boxes)
86    let tool_lines = toolbar::tool_lines(app);
87    let sym_lines = toolbar::symmetry_lines(app);
88    let blk_lines = toolbar::block_lines(app);
89    render_box_column(f, toolbar_area, &[
90        BoxContent { title: " \u{2022} Tools \u{2022} ", lines: &tool_lines },
91        BoxContent { title: " \u{2022} Symmetry \u{2022} ", lines: &sym_lines },
92        BoxContent { title: " \u{2022} Block \u{2022} ", lines: &blk_lines },
93    ], theme);
94
95    // Canvas
96    let canvas_screen_area = if app.show_preview && size.width >= 160 {
97        let side_by_side = Layout::default()
98            .direction(Direction::Horizontal)
99            .constraints([
100                Constraint::Ratio(1, 2),
101                Constraint::Ratio(1, 2),
102            ])
103            .split(canvas_area);
104        let editor_area = editor::render(f, app, side_by_side[0]);
105        editor::render_preview(f, app, side_by_side[1]);
106        editor_area
107    } else if app.show_preview {
108        editor::render_preview(f, app, canvas_area);
109        CanvasArea {
110            left: 0,
111            top: 0,
112            width: 0,
113            height: 0,
114        }
115    } else {
116        editor::render(f, app, canvas_area)
117    };
118
119    // Palette (3 boxes)
120    let colors_lines = palette::color_lines(app);
121    let section_lines = palette::section_lines(app);
122    let info_lines = palette::info_lines(app);
123    let section_title = if let Some(ref cp) = app.custom_palette {
124        format!(" \u{2022} {} \u{2022} ", cp.name)
125    } else {
126        " \u{2022} Sections \u{2022} ".to_string()
127    };
128    render_palette_column(
129        f, palette_area,
130        &colors_lines, &section_lines, &info_lines,
131        &section_title, app.palette_scroll, theme,
132    );
133
134    // Status bar (outside the border)
135    statusbar::render(f, app, status_area);
136
137    // Overlays
138    match app.mode {
139        AppMode::Help => render_help(f, app, size),
140        AppMode::Quitting => render_quit_prompt(f, size),
141        AppMode::FileDialog => render_file_dialog(f, app, size),
142        AppMode::ExportDialog => render_export_dialog(f, app, size),
143        AppMode::SaveAs => render_text_input(f, app, size, "Save As", "Enter project name:"),
144        AppMode::ExportFile => render_text_input(f, app, size, "Export", "Enter filename:"),
145        AppMode::Recovery => render_recovery_prompt(f, app, size),
146        AppMode::ColorSliders => render_color_sliders(f, app, size),
147        AppMode::PaletteDialog => render_palette_dialog(f, app, size),
148        AppMode::PaletteNameInput => render_text_input(f, app, size, "New Palette", "Enter palette name:"),
149        AppMode::PaletteRename => render_text_input(f, app, size, "Rename Palette", "Enter new name:"),
150        AppMode::PaletteExport => render_text_input(f, app, size, "Export Palette", "Enter destination path:"),
151        AppMode::NewCanvas => render_new_canvas(f, app, size),
152        _ => {}
153    }
154
155    canvas_screen_area
156}

163/// Render N bordered boxes evenly distributed vertically in a column.
164fn render_box_column(
165    f: &mut Frame,
166    column: Rect,
167    boxes: &[BoxContent],
168    theme: &Theme,
169) {
170    let n = boxes.len() as u16;
171    let box_heights: Vec<u16> = boxes.iter()
172        .map(|b| b.lines.len() as u16 + 2)
173        .collect();
174    let total_box_height: u16 = box_heights.iter().sum();
175    let remaining = column.height.saturating_sub(total_box_height);
176    let gap_count = n + 1;
177    let gap = remaining / gap_count.max(1);
178
179    let mut y = column.y + gap;
180    for (i, bx) in boxes.iter().enumerate() {
181        let h = box_heights[i];
182        let area = Rect::new(column.x, y, column.width, h);
183        render_bordered_panel(f, area, bx.lines, bx.title, theme);
184        y += h + gap;
185    }
186}

188/// Render 3 palette boxes: Colors (fixed), Sections (scrollable), Color info (fixed).
189#[allow(clippy::too_many_arguments)]
190fn render_palette_column(
191    f: &mut Frame,
192    column: Rect,
193    colors_lines: &[ratatui::text::Line<'static>],
194    section_lines: &[ratatui::text::Line<'static>],
195    info_lines: &[ratatui::text::Line<'static>],
196    section_title: &str,
197    scroll: usize,
198    theme: &Theme,
199) {
200    let colors_height = colors_lines.len() as u16 + 2;
201    let info_height = info_lines.len() as u16 + 2;
202    let gap_count = 4u16; // 3 boxes → 4 gaps
203    let section_content_height = section_lines.len() as u16;
204
205    // Sections box gets remaining space after other boxes and gaps
206    let section_max = column.height
207        .saturating_sub(colors_height + info_height + gap_count);
208    let section_box_height = (section_content_height + 2)
209        .min(section_max)
210        .max(5); // minimum 5 rows (3 headers + border)
211
212    let total_box_height = colors_height + section_box_height + info_height;
213    let remaining = column.height.saturating_sub(total_box_height);
214    let gap = remaining / gap_count.max(1);
215
216    let mut y = column.y + gap;
217
218    // Colors box
219    let colors_area = Rect::new(column.x, y, column.width, colors_height);
220    render_bordered_panel(f, colors_area, colors_lines, " \u{2022} Colors \u{2022} ", theme);
221    y += colors_height + gap;
222
223    // Sections box (scrollable)
224    let section_area = Rect::new(column.x, y, column.width, section_box_height);
225    render_bordered_panel_scrollable(f, section_area, section_lines, section_title, scroll, theme);
226    y += section_box_height + gap;
227
228    // Color info box
229    let info_area = Rect::new(column.x, y, column.width, info_height);
230    render_bordered_panel(f, info_area, info_lines, " \u{2022} Color \u{2022} ", theme);
231}

233/// Render content lines inside a vertically-centered bordered panel.
234fn render_bordered_panel(
235    f: &mut Frame,
236    column: Rect,
237    lines: &[ratatui::text::Line<'static>],
238    title: &str,
239    theme: &Theme,
240) {
241    let content_height = lines.len() as u16;
242    let panel_height = (content_height + 2).min(column.height); // +2 for border
243    let offset_y = (column.height.saturating_sub(panel_height)) / 2;
244
245    let panel_area = Rect::new(
246        column.x,
247        column.y + offset_y,
248        column.width,
249        panel_height,
250    );
251
252    let block = Block::default()
253        .borders(Borders::ALL)
254        .border_type(BorderType::Rounded)
255        .border_style(Style::default().fg(theme.border_accent))
256        .title(ratatui::text::Span::styled(
257            title.to_string(),
258            Style::default().fg(theme.border_accent).add_modifier(Modifier::BOLD),
259        ));
260
261    let paragraph = Paragraph::new(lines.to_vec()).block(block);
262    f.render_widget(paragraph, panel_area);
263}

265/// Render content lines inside a bordered panel with scroll support.
266fn render_bordered_panel_scrollable(
267    f: &mut Frame,
268    column: Rect,
269    lines: &[ratatui::text::Line<'static>],
270    title: &str,
271    scroll: usize,
272    theme: &Theme,
273) {
274    let content_height = lines.len() as u16;
275    let inner_height = column.height.saturating_sub(2); // available inside border
276
277    let (panel_area, scroll_offset) = if content_height <= inner_height {
278        // Content fits — center the panel
279        let panel_height = content_height + 2;
280        let offset_y = (column.height.saturating_sub(panel_height)) / 2;
281        (
282            Rect::new(column.x, column.y + offset_y, column.width, panel_height),
283            0u16,
284        )
285    } else {
286        // Content overflows — fill column, apply scroll
287        (column, scroll as u16)
288    };
289
290    let block = Block::default()
291        .borders(Borders::ALL)
292        .border_type(BorderType::Rounded)
293        .border_style(Style::default().fg(theme.border_accent))
294        .title(ratatui::text::Span::styled(
295            title.to_string(),
296            Style::default().fg(theme.border_accent).add_modifier(Modifier::BOLD),
297        ));
298
299    let paragraph = Paragraph::new(lines.to_vec())
300        .block(block)
301        .scroll((scroll_offset, 0));
302    f.render_widget(paragraph, panel_area);
303}

305fn render_header(f: &mut Frame, app: &App, area: Rect, theme: &Theme) {
306    let name = app
307        .project_name
308        .as_deref()
309        .unwrap_or("untitled");
310    let dirty_marker = if app.dirty { "*" } else { "" };
311    let tool_name = app.active_tool.name();
312    let sym = app.symmetry.label();
313
314    let header_text = format!(
315        " \u{0295}\u{2022}\u{1d25}\u{2022}\u{0294} kakukuma \u{2014} {}{} {:>width$}",
316        name,
317        dirty_marker,
318        format!("Tool: {}  Sym: {}", tool_name, sym),
319        width = (area.width as usize).saturating_sub(name.len() + dirty_marker.len() + 22)
320    );
321
322    let header = Paragraph::new(header_text)
323        .style(Style::default().fg(Color::White).bg(theme.header_bg));
324    f.render_widget(header, area);
325}

444fn render_file_dialog(f: &mut Frame, app: &App, area: Rect) {
445    let theme = app.theme();
446    let file_count = app.file_dialog_files.len();
447    let height = (file_count as u16 + 4).min(20);
448    let width = 44;
449    let x = (area.width.saturating_sub(width)) / 2;
450    let y = (area.height.saturating_sub(height)) / 2;
451    let dialog_area = Rect::new(x, y, width, height);
452
453    let mut lines: Vec<ratatui::text::Line> = Vec::new();
454    let visible_start = if app.file_dialog_selected > (height as usize).saturating_sub(5) {
455        app.file_dialog_selected - (height as usize).saturating_sub(5)
456    } else {
457        0
458    };
459
460    for (i, filename) in app.file_dialog_files.iter().enumerate().skip(visible_start) {
461        if lines.len() >= (height as usize).saturating_sub(4) {
462            break;
463        }
464        let is_selected = i == app.file_dialog_selected;
465        let prefix = if is_selected { "> " } else { "  " };
466        let style = if is_selected {
467            Style::default().fg(Color::Black).bg(theme.highlight)
468        } else {
469            Style::default().fg(Color::White).bg(theme.panel_bg)
470        };
471        lines.push(ratatui::text::Line::from(ratatui::text::Span::styled(
472            format!("{}{}", prefix, filename),
473            style,
474        )));
475    }
476
477    lines.push(ratatui::text::Line::from(""));
478    lines.push(ratatui::text::Line::from(ratatui::text::Span::styled(
479        " \u{2191}\u{2193} Navigate  Enter Open  Esc Cancel",
480        Style::default().fg(theme.dim).bg(theme.panel_bg),
481    )));
482
483    let dialog = Paragraph::new(lines)
484        .style(Style::default().fg(Color::White).bg(theme.panel_bg))
485        .block(
486            Block::default()
487                .borders(Borders::ALL)
488                .border_type(BorderType::Rounded)
489                .title(" Open File ")
490                .style(Style::default().fg(Color::White).bg(theme.panel_bg)),
491        );
492    f.render_widget(dialog, dialog_area);
493}

495fn render_export_dialog(f: &mut Frame, app: &App, area: Rect) {
496    let theme = app.theme();
497    let width = 40;
498    let height = 10;
499    let x = (area.width.saturating_sub(width)) / 2;
500    let y = (area.height.saturating_sub(height)) / 2;
501    let dialog_area = Rect::new(x, y, width, height);
502
503    let format_opts = ["Unicode", "ANSI Art"];
504    let dest_opts = ["Clipboard", "File"];
505
506    let mut lines: Vec<ratatui::text::Line> = Vec::new();
507
508    // Format row
509    lines.push(ratatui::text::Line::from(ratatui::text::Span::styled(
510        " Format:",
511        Style::default().fg(theme.accent).bg(theme.panel_bg),
512    )));
513    let mut fmt_spans = Vec::new();
514    fmt_spans.push(ratatui::text::Span::raw("  "));
515    for (i, opt) in format_opts.iter().enumerate() {
516        let selected = i == app.export_format;
517        let focused = app.export_cursor == 0;
518        let style = if selected && focused {
519            Style::default().fg(Color::Black).bg(theme.highlight)
520        } else if selected {
521            Style::default().fg(Color::Black).bg(Color::Gray)
522        } else {
523            Style::default().fg(Color::White).bg(theme.panel_bg)
524        };
525        fmt_spans.push(ratatui::text::Span::styled(format!(" {} ", opt), style));
526        if i == 0 {
527            fmt_spans.push(ratatui::text::Span::raw(" "));
528        }
529    }
530    lines.push(ratatui::text::Line::from(fmt_spans));
531    lines.push(ratatui::text::Line::from(""));
532
533    // Destination row
534    lines.push(ratatui::text::Line::from(ratatui::text::Span::styled(
535        " Destination:",
536        Style::default().fg(theme.accent).bg(theme.panel_bg),
537    )));
538    let mut dest_spans = Vec::new();
539    dest_spans.push(ratatui::text::Span::raw("  "));
540    for (i, opt) in dest_opts.iter().enumerate() {
541        let selected = i == app.export_dest;
542        let focused = app.export_cursor == 1;
543        let style = if selected && focused {
544            Style::default().fg(Color::Black).bg(theme.highlight)
545        } else if selected {
546            Style::default().fg(Color::Black).bg(Color::Gray)
547        } else {
548            Style::default().fg(Color::White).bg(theme.panel_bg)
549        };
550        dest_spans.push(ratatui::text::Span::styled(format!(" {} ", opt), style));
551        if i == 0 {
552            dest_spans.push(ratatui::text::Span::raw(" "));
553        }
554    }
555    lines.push(ratatui::text::Line::from(dest_spans));
556    lines.push(ratatui::text::Line::from(""));
557
558    lines.push(ratatui::text::Line::from(ratatui::text::Span::styled(
559        " \u{2191}\u{2193} Row  \u{2190}\u{2192} Option  Enter Go  Esc Cancel",
560        Style::default().fg(theme.dim).bg(theme.panel_bg),
561    )));
562
563    let dialog = Paragraph::new(lines)
564        .style(Style::default().fg(Color::White).bg(theme.panel_bg))
565        .block(
566            Block::default()
567                .borders(Borders::ALL)
568                .border_type(BorderType::Rounded)
569                .title(" Export ")
570                .style(Style::default().fg(Color::White).bg(theme.panel_bg)),
571        );
572    f.render_widget(dialog, dialog_area);
573}

631fn render_color_sliders(f: &mut Frame, app: &App, area: Rect) {
632    let theme = app.theme();
633    let width = 44;
634    let height = 14;
635    let x = (area.width.saturating_sub(width)) / 2;
636    let y = (area.height.saturating_sub(height)) / 2;
637    let dialog_area = Rect::new(x, y, width, height);
638
639    let bar_width = 20;
640    let sliders: [(&str, u16, u16); 3] = [
641        ("H", app.slider_h, 359),
642        ("S", app.slider_s as u16, 100),
643        ("L", app.slider_l as u16, 100),
644    ];
645
646    let mut lines: Vec<ratatui::text::Line> = Vec::new();
647
648    for (i, (label, value, max_val)) in sliders.iter().enumerate() {
649        let is_active = i as u8 == app.slider_active;
650        let filled = (*value as usize * bar_width) / (*max_val as usize).max(1);
651        let empty = bar_width - filled;
652        let bar: String = format!(
653            "{}{}",
654            "\u{2588}".repeat(filled),
655            "\u{2591}".repeat(empty),
656        );
657
658        let label_style = if is_active {
659            Style::default().fg(theme.accent).add_modifier(ratatui::style::Modifier::BOLD)
660        } else {
661            Style::default().fg(theme.dim)
662        };
663
664        let bar_style = if is_active {
665            Style::default().fg(Color::White).bg(theme.panel_bg)
666        } else {
667            Style::default().fg(theme.dim).bg(theme.panel_bg)
668        };
669
670        lines.push(ratatui::text::Line::from(vec![
671            ratatui::text::Span::styled(format!(" {} ", label), label_style),
672            ratatui::text::Span::styled(bar, bar_style),
673            ratatui::text::Span::styled(
674                format!(" {:>3}", value),
675                Style::default().fg(Color::White).bg(theme.panel_bg),
676            ),
677        ]));
678    }
679
680    lines.push(ratatui::text::Line::from(""));
681
682    // Live preview
683    let (r, g, b) = crate::palette::hsl_to_rgb(app.slider_h, app.slider_s, app.slider_l);
684    let preview_color = crate::palette::nearest_color(r, g, b);
685    let preview_rcolor = preview_color.to_ratatui();
686
687    lines.push(ratatui::text::Line::from(vec![
688        ratatui::text::Span::styled(" Preview: ", Style::default().fg(theme.dim).bg(theme.panel_bg)),
689        ratatui::text::Span::styled(
690            "\u{2588}\u{2588}\u{2588}\u{2588}",
691            Style::default().fg(preview_rcolor).bg(theme.panel_bg),
692        ),
693        ratatui::text::Span::styled(
694            format!("  {}", preview_color.name()),
695            Style::default().fg(theme.dim).bg(theme.panel_bg),
696        ),
697    ]));
698
699    lines.push(ratatui::text::Line::from(ratatui::text::Span::styled(
700        format!(" RGB: ({}, {}, {})  #{}", r, g, b, preview_color.0),
701        Style::default().fg(theme.dim).bg(theme.panel_bg),
702    )));
703
704    lines.push(ratatui::text::Line::from(""));
705    lines.push(ratatui::text::Line::from(ratatui::text::Span::styled(
706        " \u{2191}\u{2193} Slider  \u{2190}\u{2192} Adjust  Enter Apply  Esc Cancel",
707        Style::default().fg(theme.dim).bg(theme.panel_bg),
708    )));
709
710    let dialog = Paragraph::new(lines)
711        .style(Style::default().fg(Color::White).bg(theme.panel_bg))
712        .block(
713            Block::default()
714                .borders(Borders::ALL)
715                .border_type(BorderType::Rounded)
716                .title(" Color Sliders ")
717                .style(Style::default().fg(Color::White).bg(theme.panel_bg)),
718        );
719    f.render_widget(dialog, dialog_area);
720}

722fn render_palette_dialog(f: &mut Frame, app: &App, area: Rect) {
723    let theme = app.theme();
724    let file_count = app.palette_dialog_files.len();
725    let height = (file_count as u16 + 8).min(22);
726    let width = 44;
727    let x = (area.width.saturating_sub(width)) / 2;
728    let y = (area.height.saturating_sub(height)) / 2;
729    let dialog_area = Rect::new(x, y, width, height);
730
731    let mut lines: Vec<ratatui::text::Line> = Vec::new();
732
733    if app.palette_dialog_files.is_empty() {
734        lines.push(ratatui::text::Line::from(ratatui::text::Span::styled(
735            " No palettes found",
736            Style::default().fg(theme.dim).bg(theme.panel_bg),
737        )));
738    } else {
739        let visible_start = if app.palette_dialog_selected > (height as usize).saturating_sub(7) {
740            app.palette_dialog_selected - (height as usize).saturating_sub(7)
741        } else {
742            0
743        };
744
745        for (i, filename) in app.palette_dialog_files.iter().enumerate().skip(visible_start) {
746            if lines.len() >= (height as usize).saturating_sub(6) {
747                break;
748            }
749            let is_selected = i == app.palette_dialog_selected;
750            let prefix = if is_selected { "> " } else { "  " };
751            let style = if is_selected {
752                Style::default().fg(Color::Black).bg(theme.highlight)
753            } else {
754                Style::default().fg(Color::White).bg(theme.panel_bg)
755            };
756            lines.push(ratatui::text::Line::from(ratatui::text::Span::styled(
757                format!("{}{}", prefix, filename),
758                style,
759            )));
760        }
761    }
762
763    // Show active palette
764    if let Some(ref cp) = app.custom_palette {
765        lines.push(ratatui::text::Line::from(""));
766        lines.push(ratatui::text::Line::from(ratatui::text::Span::styled(
767            format!(" Active: {} ({} colors)", cp.name, cp.colors.len()),
768            Style::default().fg(theme.accent).bg(theme.panel_bg),
769        )));
770    }
771
772    lines.push(ratatui::text::Line::from(""));
773    lines.push(ratatui::text::Line::from(ratatui::text::Span::styled(
774        " \u{2191}\u{2193} Nav  Enter Load  N New",
775        Style::default().fg(theme.dim).bg(theme.panel_bg),
776    )));
777    lines.push(ratatui::text::Line::from(ratatui::text::Span::styled(
778        " R Rename  U Dup  D Del",
779        Style::default().fg(theme.dim).bg(theme.panel_bg),
780    )));
781    lines.push(ratatui::text::Line::from(ratatui::text::Span::styled(
782        " X Export  Esc Close",
783        Style::default().fg(theme.dim).bg(theme.panel_bg),
784    )));
785
786    let dialog = Paragraph::new(lines)
787        .style(Style::default().fg(Color::White).bg(theme.panel_bg))
788        .block(
789            Block::default()
790                .borders(Borders::ALL)
791                .border_type(BorderType::Rounded)
792                .title(" Custom Palettes ")
793                .style(Style::default().fg(Color::White).bg(theme.panel_bg)),
794        );
795    f.render_widget(dialog, dialog_area);
796}

798fn render_new_canvas(f: &mut Frame, app: &App, area: Rect) {
799    use ratatui::text::{Line, Span};
800    use ratatui::widgets::Clear;
801
802    let theme = app.theme();
803    let w = 30u16;
804    let h = 8u16;
805    let dialog_area = Rect::new(
806        area.width.saturating_sub(w) / 2,
807        area.height.saturating_sub(h) / 2,
808        w.min(area.width),
809        h.min(area.height),
810    );
811    f.render_widget(Clear, dialog_area);
812
813    let w_style = if app.new_canvas_cursor == 0 {
814        Style::default().fg(Color::Black).bg(theme.highlight).add_modifier(Modifier::BOLD)
815    } else {
816        Style::default().fg(Color::White)
817    };
818    let h_style = if app.new_canvas_cursor == 1 {
819        Style::default().fg(Color::Black).bg(theme.highlight).add_modifier(Modifier::BOLD)
820    } else {
821        Style::default().fg(Color::White)
822    };
823    let dim = Style::default().fg(theme.dim);
824
825    let lines = vec![
826        Line::from(vec![
827            Span::styled(" Width:  ", dim),
828            Span::styled(format!("\u{25C0} {:>3} \u{25B6}", app.new_canvas_width), w_style),
829        ]),
830        Line::from(Span::raw("")),
831        Line::from(vec![
832            Span::styled(" Height: ", dim),
833            Span::styled(format!("\u{25C0} {:>3} \u{25B6}", app.new_canvas_height), h_style),
834        ]),
835        Line::from(Span::raw("")),
836        Line::from(Span::styled(" Enter=Create  Esc=Cancel", dim)),
837    ];
838
839    let dialog = Paragraph::new(lines).block(
840        Block::default()
841            .borders(Borders::ALL)
842            .border_type(BorderType::Rounded)
843            .title(" New Canvas ")
844            .style(Style::default().fg(theme.accent).bg(theme.panel_bg)),
845    );
846    f.render_widget(dialog, dialog_area);
