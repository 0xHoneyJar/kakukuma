9//! Aarch64 color conversion routines
10//! NEON is mandatory on aarch64.
11
12#![cfg(all(feature = "neon", target_arch = "aarch64"))]
13use std::arch::aarch64::*;
14
15use crate::color_convert::scalar::{CB_CF, CR_CF, C_G_CB_COEF_2, C_G_CR_COEF_1, YUV_RND, Y_CF};
16
17const C_1: u64 = u64::from_ne_bytes([
18    Y_CF.to_ne_bytes()[0],
19    Y_CF.to_ne_bytes()[1],
20    CR_CF.to_ne_bytes()[0],
21    CR_CF.to_ne_bytes()[1],
22    CB_CF.to_ne_bytes()[0],
23    CB_CF.to_ne_bytes()[1],
24    C_G_CR_COEF_1.to_ne_bytes()[0],
25    C_G_CR_COEF_1.to_ne_bytes()[1]
26]);
27const C_2: u64 = u64::from_ne_bytes([
28    C_G_CB_COEF_2.to_ne_bytes()[0],
29    C_G_CB_COEF_2.to_ne_bytes()[1],
30    0,
31    0,
32    0,
33    0,
34    0,
35    0
36]);
37
38#[inline(always)]
39unsafe fn ycbcr_to_rgb_baseline_no_clamp(
40    y: &[i16; 16], cb: &[i16; 16], cr: &[i16; 16]
41) -> (uint8x16_t, uint8x16_t, uint8x16_t) {
42    // NEON has 32 registers, so it is good idea to utilize a lot of variables at once
43
44    let cb_cr_bias = vdupq_n_s16(128);
45    // 0 - Y coeff, 1 - Cr, 2 - Cb, 3 - G1, 4 - G2
46    let coefficients = vcombine_s16(vcreate_s16(C_1), vcreate_s16(C_2));
47
48    let y0 = vld1q_s16(y.as_ptr().cast());
49    let y1 = vld1q_s16(y[8..].as_ptr().cast());
50
51    let mut cb0 = vld1q_s16(cb.as_ptr().cast());
52    let mut cb1 = vld1q_s16(cb[8..].as_ptr().cast());
53
54    let mut cr0 = vld1q_s16(cr.as_ptr().cast());
55    let mut cr1 = vld1q_s16(cr[8..].as_ptr().cast());
56
57    cb0 = vsubq_s16(cb0, cb_cr_bias);
58    cb1 = vsubq_s16(cb1, cb_cr_bias);
59
60    cr0 = vsubq_s16(cr0, cb_cr_bias);
61    cr1 = vsubq_s16(cr1, cb_cr_bias);
62
63    let bias = vdupq_n_s32(i32::from(YUV_RND));
64
65    let acc0 = vmlal_laneq_s16::<0>(bias, vget_low_s16(y0), coefficients);
66    let acc1 = vmlal_high_laneq_s16::<0>(bias, y0, coefficients);
67    let acc2 = vmlal_laneq_s16::<0>(bias, vget_low_s16(y1), coefficients);
68    let acc3 = vmlal_high_laneq_s16::<0>(bias, y1, coefficients);
69
70    let r0 = vmlal_laneq_s16::<1>(acc0, vget_low_s16(cr0), coefficients);
71    let r1 = vmlal_high_laneq_s16::<1>(acc1, cr0, coefficients);
72    let r2 = vmlal_laneq_s16::<1>(acc2, vget_low_s16(cr1), coefficients);
73    let r3 = vmlal_high_laneq_s16::<1>(acc3, cr1, coefficients);
74
75    let b0 = vmlal_laneq_s16::<2>(acc0, vget_low_s16(cb0), coefficients);
76    let b1 = vmlal_high_laneq_s16::<2>(acc1, cb0, coefficients);
77    let b2 = vmlal_laneq_s16::<2>(acc2, vget_low_s16(cb1), coefficients);
78    let b3 = vmlal_high_laneq_s16::<2>(acc3, cb1, coefficients);
79
80    // Saturating shift right with signed -> unsigned saturation
81    let qr0 = vqshrun_n_s32::<14>(r0);
82    let qr1 = vqshrun_n_s32::<14>(r1);
83    let qr2 = vqshrun_n_s32::<14>(r2);
84    let qr3 = vqshrun_n_s32::<14>(r3);
85
86    let mut g0 = vmlal_laneq_s16::<4>(acc0, vget_low_s16(cb0), coefficients);
87    let mut g1 = vmlal_high_laneq_s16::<4>(acc1, cb0, coefficients);
88    let mut g2 = vmlal_laneq_s16::<4>(acc2, vget_low_s16(cb1), coefficients);
89    let mut g3 = vmlal_high_laneq_s16::<4>(acc3, cb1, coefficients);
90
91    let qb0 = vqshrun_n_s32::<14>(b0);
92    let qb1 = vqshrun_n_s32::<14>(b1);
93    let qb2 = vqshrun_n_s32::<14>(b2);
94    let qb3 = vqshrun_n_s32::<14>(b3);
95
96    let r0 = vqmovn_u16(vcombine_u16(qr0, qr1));
97    let r1 = vqmovn_u16(vcombine_u16(qr2, qr3));
98
99    let b0 = vqmovn_u16(vcombine_u16(qb0, qb1));
100    let b1 = vqmovn_u16(vcombine_u16(qb2, qb3));
101
102    g0 = vmlal_laneq_s16::<3>(g0, vget_low_s16(cr0), coefficients);
103    g1 = vmlal_high_laneq_s16::<3>(g1, cr0, coefficients);
104    g2 = vmlal_laneq_s16::<3>(g2, vget_low_s16(cr1), coefficients);
105    g3 = vmlal_high_laneq_s16::<3>(g3, cr1, coefficients);
106
107    let qg0 = vqshrun_n_s32::<14>(g0);
108    let qg1 = vqshrun_n_s32::<14>(g1);
109    let qg2 = vqshrun_n_s32::<14>(g2);
110    let qg3 = vqshrun_n_s32::<14>(g3);
111
112    let g0 = vqmovn_u16(vcombine_u16(qg0, qg1));
113    let g1 = vqmovn_u16(vcombine_u16(qg2, qg3));
114
115    (
116        vcombine_u8(r0, r1),
117        vcombine_u8(g0, g1),
118        vcombine_u8(b0, b1)
119    )
120}

122#[inline(always)]
123pub fn ycbcr_to_rgb_neon(
124    y: &[i16; 16], cb: &[i16; 16], cr: &[i16; 16], out: &mut [u8], offset: &mut usize
125) {
126    // call this in another function to tell RUST to vectorize this
127    // storing
128    unsafe {
129        let (r, g, b) = ycbcr_to_rgb_baseline_no_clamp(y, cb, cr);
130        vst3q_u8(out.as_mut_ptr(), uint8x16x3_t(r, g, b));
131        *offset += 48;
132    }
133}
134
135#[inline(always)]
136pub fn ycbcr_to_rgba_neon(
137    y: &[i16; 16], cb: &[i16; 16], cr: &[i16; 16], out: &mut [u8], offset: &mut usize
138) {
139    unsafe {
140        let (r, g, b) = ycbcr_to_rgb_baseline_no_clamp(y, cb, cr);
141        vst4q_u8(out.as_mut_ptr(), uint8x16x4_t(r, g, b, vdupq_n_u8(255)));
142        *offset += 64;
143    }
144}
